using System;

    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Enter your choice of conversion:\n1) Celsius to Fahrenheit or 2) Fahrenheit to Celsius\nYour choice = ");
            double choice = double.Parse(Console.ReadLine());
            if (choice == 1)
            {
                Console.WriteLine("Enter temperature in Celsius: ");
                double celsius = double.Parse(Console.ReadLine());

                double fahrenheit = (celsius * 9) / 5 + 32;
                Console.WriteLine("Temperature in Fahrenheit: " + fahrenheit);
            }
            else if (choice == 2)
            {
                Console.WriteLine("Enter temperature in fahrenheit: ");
                double fahrenheit = double.Parse(Console.ReadLine());

                double celsius = (fahrenheit - 32) / 1.8;
                Console.WriteLine("Temperature in Celsius: " + celsius);
            }
            else { Console.WriteLine("Please enter a valid choice"); }
        }
    }
