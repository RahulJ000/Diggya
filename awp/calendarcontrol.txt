    {

        {

        }

        protected void Calendar1_SelectionChanged(object sender, EventArgs e)
        {

        }

        protected void Calendar1_DayRender(object sender, DayRenderEventArgs e)
        {
            if (e.Day.Date.Day == 31)
                e.Cell.Controls.Add(new LiteralControl("</br>HalloweenðŸŽƒ"));

            if ((e.Day.Date >= new DateTime(2022, 10, 9)) && (e.Day.Date <= new DateTime(2022, 10, 15)))
            {
                e.Cell.BackColor = System.Drawing.Color.Cyan;
                e.Cell.BorderColor = System.Drawing.Color.Black;
                e.Cell.BorderWidth = new Unit(3);
                if (e.Day.IsOtherMonth)
                {
                    e.Cell.Controls.Clear();
                }
            }
        }

        protected void Button1_Click(object sender, EventArgs e)
        {
            TimeSpan t = Calendar2.SelectedDate - Calendar1.SelectedDate;
            Label1.Text += t.Days.ToString();

        }
    }
